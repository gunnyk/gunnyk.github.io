function cloneObj(obj) {
  let cloneObject = {};
  
  for (let key in obj) {
    if (typeof obj[key] == 'object' && obj[key] != null) {
      cloneObject[key] = cloneObj(obj[key]);
    } else {
      cloneObject[key] = obj[key];
    }
  }

  return cloneObject;
}
